// 从左右两边 两遍最长递增子序列
// 这个不是这样写的
#include <iostream>
#include <iomanip>
#include <cmath>
#include <stdio.h>
#include <queue>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;


int main(){
	int n;
	cin >> n;
	int t[110], up[110], down[110];
	int sum=0;
	int pre=-1;
	for (int i = 0; i < n; i++){
		int temp;
		cin >> temp;
		if (temp == pre) continue;
		else{
			t[sum++]=temp;
			pre=temp;
		}
	}
	for (int i = 0; i < sum; i++){
		int up_max = 1;
		int down_max = 1;
		for (int j = i-1; j >= 0; j--){
			if (t[i] > t[j])
				up_max = max(up_max, up[j]+1);
			else if (t[i] == t[j]){
				continue;
			}
			else{
				down_max = max(down_max, down[j]+1);
				if (up[j] + 1 > down_max)
					down_max = up[j] + 1;
			}
		}
		up[i] = up_max;
		down[i] = down_max;
	}

	int max_num = -1;
	for (int i = 0; i < sum; i++){
		if (up[i] > max_num)
			max_num = up[i];
		if (down[i] > max_num)
			max_num = down[i];
	}
	cout << n-max_num << endl;
    return 0;
}
