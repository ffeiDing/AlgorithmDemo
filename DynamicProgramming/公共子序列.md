// 如果i,j两个位置对应元素相等，要么len(i,j)+1，要么len(i-1,j)，要么len(i,j-1)
#include <iostream>
#include <iomanip>
#include <cmath>
#include <stdio.h>
#include <queue>
#include <cstring>
#include <algorithm>
#include <set>
using namespace std;

int main(){
	string a, b;
	while(cin >> a >> b){
		int a_len = a.length();
		int b_len = b.length();
		int len[210][210];
		memset(len, 0, sizeof(len));
		for (int i = 1; i <= a_len; i++){
			for (int j = 1; j <= b_len; j++){
				if (a[i-1] == b[j-1])
					len[i][j] = max(len[i-1][j-1]+1, max(len[i-1][j], len[i][j-1]));
				else
					len[i][j] = max(len[i-1][j], len[i][j-1]);
			}
		}
		cout << len[a_len][b_len] << endl;
	}
    return 0;
}
