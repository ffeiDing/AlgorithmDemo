// 0-1背包问题
#include <iostream>
#include <iomanip>
#include <cmath>
#include <stdio.h>
#include <queue>
#include <cstring>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int t, m;
int w[110], v[110];
int sum[110][1010];

int main(){
	scanf("%d%d", &t, &m);
	for (int i = 0; i < m; i++)
		scanf("%d%d", &w[i], &v[i]);
	memset(sum[0], 0, sizeof(sum[0]));
	for (int i = 1; i <= m; i++){
		for (int j = 0; j <= t; j++){
			sum[i][j] = sum[i-1][j];
			if (j >= w[i-1])
				sum[i][j] = max(sum[i][j], sum[i-1][j-w[i-1]]+v[i-1]);
		}
	}
	int res = -1;
	for (int i = 0; i <= t; i++)
		res = max(res, sum[m][i]);
	printf("%d\n", res);
    return 0;
}

// 0-1背包从后向前遍历，完全背包从前向后遍历
#include <iostream>
#include <iomanip>
#include <cmath>
#include <stdio.h>
#include <queue>
#include <cstring>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int t, m;
int w[110], v[110];
int sum[1010];

int main(){
	scanf("%d%d", &t, &m);
	for (int i = 0; i < m; i++)
		scanf("%d%d", &w[i], &v[i]);
	memset(sum, 0, sizeof(sum));
	for (int i = 0; i < m; i++){
		for (int j = t; j >= w[i]; j--){
			sum[j] = max(sum[j], sum[j-w[i]]+v[i]);
		}
	}
	int res = -1;
	for (int i = 0; i <= t; i++)
		res = max(res, sum[i]);
	printf("%d\n", res);
    return 0;
}
