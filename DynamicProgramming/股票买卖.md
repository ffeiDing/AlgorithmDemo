# 一、单次股票买卖

## O(n^2)解法
```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        int res_max = 0;
        for (int i = 0; i < len; i++){
            int max_num = prices[i];
            for (int j = i; j < len; j++){
                max_num = max(max_num, prices[j]);
            }
            res_max = max(res_max, max_num - prices[i]);
        }
        return res_max;
    }
};
```

## O(n)解法
```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int max_res = 0;
        int len = prices.size();
        if (len == 0) return max_res;
        int now_num = prices[0];
        for (int j = 1; j < len; j++){
            if (prices[j] > now_num)
                max_res = max(max_res, prices[j]-now_num);
            else
                now_num = prices[j];
        }
        return max_res;
    }
};
```

# 二、无限次股票买卖

```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int max_res = 0;
        int len = prices.size();
        for (int i = 0; i < len - 1; i++){
            if (prices[i+1] > prices[i])
                max_res += prices[i+1] - prices[i];
        }
        return max_res;
    }
};
```
