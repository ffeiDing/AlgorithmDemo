# 一、单次股票买卖

## O(n^2)解法
```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int len = prices.size();
        int res_max = 0;
        for (int i = 0; i < len; i++){
            int max_num = prices[i];
            for (int j = i; j < len; j++){
                max_num = max(max_num, prices[j]);
            }
            res_max = max(res_max, max_num - prices[i]);
        }
        return res_max;
    }
};
```

## O(n)解法
```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int max_res = 0;
        int len = prices.size();
        if (len == 0) return max_res;
        int now_num = prices[0];
        for (int j = 1; j < len; j++){
            if (prices[j] > now_num)
                max_res = max(max_res, prices[j]-now_num);
            else
                now_num = prices[j];
        }
        return max_res;
    }
};
```

# 二、无限次股票买卖

```c
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int max_res = 0;
        int len = prices.size();
        for (int i = 0; i < len - 1; i++){
            if (prices[i+1] > prices[i])
                max_res += prices[i+1] - prices[i];
        }
        return max_res;
    }
};
```
# 三、两次股票买卖

#include <iostream>

int t;
int prices[100010];
int fp[100010];
int bp[100010];
int n;

int main(){
	scanf("%d", &t);
	while(t--){
		scanf("%d", &n);
		for (int i = 0; i < n; i++)
			scanf("%d", &prices[i]);
		memset(fp, 0, sizeof(fp));
		memset(bp, 0, sizeof(bp));
		int max_res = 0;
		int now_num = prices[0];
		int max_num = 0;
		for (int i = 0; i < n; i++){
			if (prices[i] > now_num)
				max_num = max(max_num, prices[i]-now_num);
			else
				now_num = prices[i];
			fp[i] = max_num;
		}
		now_num = prices[n-1];
		max_num = 0;
		for (int i = n-1; i >= 0; i--){
			if (prices[i] < now_num)
				max_num = max(max_num, now_num-prices[i]);
			else
				now_num = prices[i];
			bp[i] = max_num;
		}
		for (int i = 0; i < n; i++){
			max_res = max(max_res, fp[i]+bp[i]);
		}
		printf("%d\n", max_res);
	}
    return 0;
}
