#include <iostream>
#include <iomanip>
#include <cmath>
#include <map>
#include <cstring>
using namespace std;

struct Node{
	int leaf[110];
	int in;
	int out;
};
Node node[110];
int vis[110];
int n, m;

int main(){
	cin >> n >> m;
	memset(vis, 0, sizeof(vis));
	for (int i = 1; i <= n; i++){
		node[i].in = 0;
		node[i].out = 0;
	}
	for (int i = 0; i < m; i++){
		int a, b;
		cin >> a >> b;
		node[a].leaf[node[a].out] = b;
		node[a].out++;
		node[b].in++;
	}
	int count = 0;
	int flag = 0;
	while(count < n){
		for (int i = 1; i <= n; i++){
			if (vis[i] == 0 && node[i].in == 0){
				if (flag == 1)
					cout << " ";
				else
					flag = 1;
				cout << "v" << i;
				for (int j = 0; j < node[i].out; j++){
					node[node[i].leaf[j]].in--;
				}
				vis[i] = 1;
				count++;
				break;
			}
		}
	}
	cout << endl;
	return 0;
}
