//动态规划
class Solution {
public:
    bool isMatch(string s, string p) {
        int len1 = s.length();
        int len2 = p.length();
        bool dp[len1+1][len2+1];
        dp[len1][len2] = true;
        for (int i = len1; i>=0; i--){
            for (int j = len2; j>=0; j--){
                if (i == len1 && j == len2)
                    continue;
                bool ij = s[i]==p[j] || p[j] == '.';
                if (j < len2 && p[j+1] == '*' 
                    || i == len1 && p[j+1] == '*'){
                    if (!ij || i == len1)
                        dp[i][j] = dp[i][j+2];
                    else
                        dp[i][j] = dp[i+1][j+2] | dp[i][j+2] | dp[i+1][j];
                    continue;
                }
                if (ij && j < len2 && i < len1)
                    dp[i][j] = dp[i+1][j+1];
                else
                    dp[i][j] = false;
            }
        }
        return dp[0][0];
    }
};
