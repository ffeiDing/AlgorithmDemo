class Solution {
public:
    bool cnt(vector<string>& board, char c){
        int n = board.size();
        for (int i = 0; i < n; i++){
            int len = 0;
            for (int j = 0; j < n; j++){
                if (board[i][j] == c){
                    len++;
                    if (len == n)
                        return true;
                }
            }
        }
        for (int j = 0; j < n; j++){
            int len = 0;
            for (int i = 0; i < n; i++){
                if (board[i][j] == c){
                    len++;
                    if (len == n)
                        return true;
                }
            }
        }
        for (int sum = 0; sum <= 2*n-2; sum++){
            int len = 0;
            for (int i = 0; i < n; i++){
                int j = sum-i;
                if (j < 0 || j >= n) continue;
                if (board[i][j] == c){
                    len++;
                    if (len == n)
                        return true;
                }
            }
        }
        for (int sub = 1-n; sub <= n-1; sub++){
            int len = 0;
            for (int i = 0; i < n; i++){
                int j = i-sub;
                if (j < 0 || j >= n) continue;
                if (board[i][j] == c){
                    len++;
                    if (len == n)
                        return true;
                }
            }
        }
        return false;
    }
    string tictactoe(vector<string>& board) {
        int n = board.size();
        int empty = 0;
        for (int i = 0; i < n; i++){
            for (int j = 0; j < n; j++){
                if (board[i][j] == ' '){
                    empty = 1;
                    break;
                }
            }
        }
        if (cnt(board, 'O')) return "O";
        if (cnt(board, 'X')) return "X";
        if (empty) return "Pending";
        return "Draw";
    }
};
