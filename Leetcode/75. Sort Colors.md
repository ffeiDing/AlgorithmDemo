class Solution {
public:
    void sortColors(vector<int>& nums) {
        int len = nums.size();
        int a = -1;
        int b = -1;
        int c = -1;
        for (int i = 0; i < len; i++){
            if (nums[i] == 0){
                nums[++a] = 2;
                nums[++b] = 1;
                nums[++c] = 0;
            }
            else if (nums[i] == 1){
                nums[++a] = 2;
                nums[++b] = 1;
            }
            else{
                nums[++a] = 2;
            }
        }
    }
};

class Solution {
public:
    void sortColors(vector<int>& nums) {
        int len = nums.size();
        int l = 0;
        int r = len-1;
        for (int i = 0; i <= r; i++){
            if (nums[i] == 0){
                int tmp = nums[l];
                nums[l] = nums[i];
                nums[i] = tmp;
                l++;
            }
            else if (nums[i] == 2){
                int tmp = nums[r];
                nums[r] = nums[i];
                nums[i] = tmp;
                i--;
                r--;
            }
        }
    }
};
