class Solution {
public:
    int hammingDistance(int x, int y) {
        int res = 0;
        while(x > 0 && y > 0){
            res += x%2 == y%2 ? 0:1;
            x = x>>1;
            y = y>>1;
        }
        if (y > 0)
            x = y;
        while(x > 0){
            res += x%2;
            x = x>>1;
        }
        return res;
    }
};
