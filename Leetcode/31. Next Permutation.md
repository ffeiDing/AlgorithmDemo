class Solution {
public:
    void nextPermutation(vector<int>& nums) {
        int len = nums.size();
        int left = len-2;
        while(left >= 0){
            if (nums[left] < nums[left+1])
                break;
            left--;
        }
        if (left == -1){
            sort(nums.begin(), nums.end());
            return;
        }
        int right = -1;
        int min_num = 10010;
        for (int i = len-1; i > left; i--){
            if (nums[i] < min_num && nums[i] > nums[left]){
                min_num = nums[i];
                right = i;
            }
        }
        int temp = nums[right];
        nums[right] = nums[left];
        nums[left] = temp;
        sort(nums.begin()+left+1, nums.end());
        return;
    }
};
