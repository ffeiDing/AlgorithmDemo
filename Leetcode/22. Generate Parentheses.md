class Solution {
public:
    vector<string> generateParenthesis(int n) {
        vector<vector<string>> ga;
        vector<string> l;
        l.push_back("");
        ga.push_back(l);
        for (int i = 1; i <= n; i++){
            vector<string> s;
            for (int j = 0; j < i; j++){
                for (int k = 0; k < ga[j].size(); k++){
                    for (int h = 0; h < ga[i-1-j].size(); h++){
                        string tmp = "(" + ga[j][k] + ")" + ga[i-1-j][h];
                        s.push_back(tmp);
                    }
                }
            }
            ga.push_back(s);
        }
        return ga[n];
    }
};
