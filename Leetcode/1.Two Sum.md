class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        int len = nums.size();
        int a, b;
        for (int i = 0; i < len; i++){
            a = i;
            int temp = nums[i];
            cout << temp << endl;
            for (int j = i+1; j < len; j++){
                if (temp + nums[j] == target){
                    b = j;
                    vector<int> res;
                    res.push_back(a);
                    res.push_back(b);
                    return res; 
                }
            }
        }
        vector<int> res;
        return res; 
    }
};
