```
class Solution {
public:
    int numDecodings(string s) {
        int len = s.length();
        s += 'x';
        int dp[10010];
        dp[len] = 1;
        for (int i = len-1; i >= 0; i--){
            if (s[i] == '0') dp[i] = 0;
            else{
                int tmp = (s[i]-'0')*10+(s[i+1]-'0');
                if (tmp <= 26 && tmp >= 10)
                    dp[i] = dp[i+1]+dp[i+2];
                else
                    dp[i] = dp[i+1];
            }
        }
        return dp[0];
    }
};
```
