/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        ListNode* head(0);
        if (l1 == NULL && l2 == NULL)
            return NULL;
        if (l1 == NULL)
            return l2;
        if (l2 == NULL)
            return l1;
        if (l1->val <= l2->val)
            head = l1;
        else
            head = l2;
        while(l1 != NULL && l2 != NULL){
            if (l1->val <= l2->val){
                while (l1->next != NULL && l1->next->val <= l2->val)
                    l1 = l1->next;
                ListNode* tmp = l1->next;
                l1->next = l2;
                l1 = tmp;
            }
            else{
                while (l2->next != NULL && l2->next->val < l1->val)
                    l2 = l2->next;
                ListNode* tmp = l2->next;
                l2->next = l1;
                l2 = tmp;
            }
        }
        return head;
    }
};
