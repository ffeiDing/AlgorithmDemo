class Solution {
public:
    int findK(vector<int>& nums, int k, int s, int e){
        int l = s;
        int r = e;
        int privot = nums[l];
        if (l == r) return nums[l];
        while(l < r){
            while(l < r && nums[r] <= privot) r--;
            nums[l] = nums[r];
            while(l < r && nums[l] >= privot) l++;
            nums[r] = nums[l];
        }
        nums[l] = privot;
        cout << l << endl;
        if (l-s == k) return privot;
        if (l-s > k)
            return findK(nums, k, s, l);
        else
            return findK(nums, k-l+s-1, l+1, e);
    }
    int findKthLargest(vector<int>& nums, int k) {
        int n = nums.size();
        return findK(nums, k-1, 0, n-1);
    }
};

class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        int vis[10010];
        int n = nums.size();
        memset(vis, 0, sizeof(vis));
        int res;
        for (int i = 0; i < k; i++){
            res = -10010;
            int idx = -1;
            for (int j = 0; j < n; j++){
                if (vis[j] == 1) continue;
                if (nums[j] > res){
                    res = nums[j];
                    idx = j;
                }
            }
            vis[idx] = 1;
        }
        return res;
    }
};
