class Node{
public:
    bool is_leaf;
    Node* nodes[26];
    Node(){
        is_leaf = false;
        memset(nodes, NULL, sizeof(nodes));
    }
};

class WordDictionary {
public:
    /** Initialize your data structure here. */
    Node* root;
    WordDictionary() {
        root = new Node();
    }
    
    void addWord(string word) {
        int len = word.length();
        Node* node = root;
        for (int i = 0; i < len; i++){
            int tmp = word[i]-'a';
            if (node->nodes[tmp] == NULL){
                node->nodes[tmp] = new Node();
            }
            node = node->nodes[tmp];
        }
        node->is_leaf = true;
    }
    
    bool search_tree(string word, Node* root){
        int len = word.length();
        int idx = 0;
        Node* node = root;
        while (idx < len && node){
            if (word[idx] == '.'){
                for (int i = 0; i < 26; i++){
                    if (search_tree(word.substr(idx+1), node->nodes[i]))
                        return true;
                }
                return false;
            }
            else{
                node = node->nodes[word[idx]-'a'];
                idx++;
            }            
        }
        return node && node->is_leaf;
    }
    
    bool search(string word) {
        Node* node = root;
        return search_tree(word, root);        
    }
};

/**

 * WordDictionary* obj = new WordDictionary();
 * obj->addWord(word);
 * bool param_2 = obj->search(word);
