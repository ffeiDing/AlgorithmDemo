class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        if (m == 0) return false;
        int n = matrix[0].size();
        if (n == 0) return false;
        int i = 0;
        int j = n-1;
        while(1){
            while(matrix[i][j] > target){
                j--;
                if (j < 0) return false;
            }
            while (matrix[i][j] < target){
                i++;
                if (i >= m) return false;
            }
            if (matrix[i][j] == target)
                return true;
        }
        return false;        
    }
};

class Solution {
public:
    bool searchTarget(vector<vector<int>>& matrix, int l1, int l2, int r1, int r2, int target){
        if (l1 == r1 && l2 == r2){
            if (matrix[l1][l2] == target) return true;
            return false;
        }
        int mid1 = (l1+r1)/2;
        int mid2 = (l2+r2)/2;
        if (matrix[mid1][mid2] == target) return true;
        if (matrix[mid1][mid2] < target){
            bool tmp1 = false;
            bool tmp2 = false;
            bool tmp3 = false;
            if (mid2+1 <= r2)
                tmp1 = searchTarget(matrix, l1, mid2+1, mid1, r2, target);
            if (tmp1) return tmp1;
            if (mid1+1 <= r1)
                tmp2 = searchTarget(matrix, mid1+1, l2, r1, mid2, target);
            if (tmp2) return tmp2;
            if (mid1+1 <= r1 && mid2+1 <= r2)
                tmp3 = searchTarget(matrix, mid1+1, mid2+1, r1, r2, target);
            if (tmp3) return tmp3;
            return false;
        }
        if (matrix[mid1][mid2] > target){
            bool tmp1 = false;
            bool tmp2 = false;
            bool tmp3 = false;
            if (mid2-1 >= l2)
                tmp1 = searchTarget(matrix, mid1, l2, r1, mid2-1, target);
            if (tmp1) return tmp1;
            if (mid1-1 >= l1)
                tmp2 = searchTarget(matrix, l1, mid2, mid1-1, r2, target);
            if (tmp2) return tmp2;
            if (mid1-1 >= l1 && mid2-1 >= l2)
                tmp3 = searchTarget(matrix, l1, l2, mid1-1, mid2-1, target);
            if (tmp3) return tmp3;
            return false;
        }
        return false;
    }
    
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        int m = matrix.size();
        if (m == 0) return false;
        int n = matrix[0].size();
        if (n == 0) return false;
        return searchTarget(matrix, 0, 0, m-1, n-1, target);
    }
};
