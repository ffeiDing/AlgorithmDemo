```
class Solution {
public:
    int subarrayBitwiseORs(vector<int>& arr) {
        unordered_set<int> cur, res, tmp;
        for (auto num : arr){
            tmp = {num};
            for (auto i : cur)
                tmp.insert(num|i);
            cur = tmp;
            for (auto i : cur)    
                res.insert(i);
        }
        return res.size();
    }
};
```
