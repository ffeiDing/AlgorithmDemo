class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        if (k == 0) return;
        int n = nums.size();
        int cnt = 0;
        for(int start = 0; cnt < n; start++){
            int idx = start;
            int pre = nums[start];
            do{
                int next = (idx+k)%n;
                int tmp = nums[next];
                nums[next] = pre;
                pre = tmp;
                idx = next;
                cnt++;
            }while(start != idx);
            idx = (idx+1)%n;
            pre = nums[idx];
        }
    }
};
