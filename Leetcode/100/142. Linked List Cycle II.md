/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode *tmp = new ListNode(0);
        ListNode *nxt = new ListNode(0);
        tmp = head;
        nxt = head;
        if (head == NULL) return NULL;
        while(tmp->next != NULL){   
            tmp = tmp->next;
            if (nxt->next == NULL)
                return NULL;
            if (nxt->next->next == NULL)
                return NULL;
            nxt = nxt->next->next;
            if (tmp == nxt){
                tmp = head;
                while(tmp != nxt){
                    tmp = tmp->next;
                    nxt = nxt->next;
                }
                return tmp;
            }
        }
        return NULL;
    }
};
