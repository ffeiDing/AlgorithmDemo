class LRUCache {
public:
    int n;
    list<pair<int, int>> lis;
    unordered_map<int, list<pair<int, int>>::iterator> m;
    LRUCache(int capacity) {
        n = capacity;
    }
    
    int get(int key) {
        auto it = m.find(key);
        if (it == m.end()) return -1;
        int res = it->second->second;
        lis.erase(it->second);
        lis.push_front(make_pair(key, res));
        m[key] = lis.begin();
        return res;
    }
    
    void put(int key, int value) {
        auto it = m.find(key);
        if (it != m.end()){
            auto upd = it->second;
            lis.erase(upd);
            lis.push_front(make_pair(key, value));
            m[key] = lis.begin();
            return;
        }
        if (m.size() >= n){
            auto aban = lis.end();
            aban--;
            m.erase(aban->first);
            lis.erase(aban);
        }
        lis.push_front(make_pair(key, value));
        m[key] = lis.begin();
        return;
    }
};

 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
