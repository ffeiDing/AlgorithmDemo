class Solution {
public:
    int candy(vector<int>& ratings) {
        int len = ratings.size();
        int res = 0;
        if (len == 0) return res;
        int num[100010];
        num[0] = 1;
        for (int i = 1; i < len; i++){
            if (ratings[i] > ratings[i-1])
                num[i] = num[i-1]+1;
            else
                num[i] = 1;
        }
        res = num[len-1];
        for (int i = len-2; i >= 0; i--){
            if (ratings[i] > ratings[i+1])
                num[i] = max(num[i+1]+1, num[i]);
            //cout << num[i] << endl;
            res += num[i];
        }
        return res;
    }
};
