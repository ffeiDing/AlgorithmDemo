class Solution {
public:
    int majorityElement(vector<int>& nums) {
        int len = nums.size();
        int cnt = 0;
        int res;
        for (int i = 0; i < len; i++){
            if (cnt == 0) res = nums[i];
            if (nums[i] == res) cnt++;
            else cnt--;
        }
        return res;
    }
};
