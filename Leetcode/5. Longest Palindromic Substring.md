class Solution {
public:
    int len_palindromic(string s, int n, int len){
        //cout << s << endl;
        int i = 1;
        int re = 0;
        while(1){
            //cout << i << " " << re << endl;
            if (n+i > len || n-i < 0)
                return i-re;
            if (s[n+i] != s[n-i])
                return i-re;
            if (s[n+i] == ' ' && s[n-i] == ' '){
                i++;
                re++;
                continue;
            }
            if (s[n+i] == s[n-i]){
                i++;
                continue;
            }
        }
    }
    string longestPalindrome(string s) {
        int len = s.length();
        if (len == 0)
            return "";
        int num[2020];
        char str[2020];
        for (int i = 0; i < len; i++){
            str[i*2] = s[i];
            str[i*2+1] = ' ';
        }
        str[len*2] = '\0';
        //cout << str << endl;
        string s_new = str;    
        int max_num = -1;
        int max_idx = -1;
        for (int i = 0; i < len*2; i++){
            num[i] = len_palindromic(s_new, i, len*2);
            if (str[i] == ' ')
                num[i] = num[i]*2-2;
            else
                num[i] = num[i]*2-1;
            if (num[i] > max_num){
                max_num = num[i];
                max_idx = i;
            }
        }
        max_num = max_num/2;
        char res[2020];
        if (str[max_idx] == ' '){
            int idx = 1;
            for (int i = max_idx/2; i >= max_idx/2 - max_num + 1; i--){
                res[max_num-idx] = s[i];
                res[max_num+idx-1] = s[i];
                idx++;
            } 
            res[max_num*2] = '\0';
        }
        else{
            max_num++;
            int idx = 0;
            for (int i = max_idx/2; i >= max_idx/2 - max_num + 1; i--){
                res[max_num-idx-1] = s[i];
                res[max_num+idx-1] = s[i];
                idx++;
            }
            res[max_num*2-1] = '\0';
        }
        string res_new = res;
        return res_new;
    }
};
