class Solution {
public:
    int trap(vector<int>& height) {
        int len = height.size();
        stack<int> s;
        int res = 0;
        for (int i = 0; i < len; i++){
            if (s.empty() || height[i] <= height[s.top()])
                s.push(i);
            else{
                while(!s.empty() && height[s.top()]<height[i]){
                    int temp = s.top();
                    s.pop();
                    if (!s.empty()){
                        int l = s.top();
                        int h = min(height[l],height[i]) - height[temp];
                        int w = i-l-1;
                        res += h*w;
                    }
                }
                s.push(i);
            }
        }
        return res;
    }
};
