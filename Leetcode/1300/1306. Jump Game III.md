class Solution {
public:
    int len;
    int vis[50010];
    bool is_valid(int a){
        return a >= 0 && a < len;
    }
    bool canReach(vector<int>& arr, int start) {
        int tmp = arr[start];
        if (tmp == 0)
            return true;
        len = arr.size();
        vis[start] = 1;
        int a = start-tmp;
        int b = start+tmp;
        bool l = false;
        bool r = false;
        if (is_valid(a) && vis[a] == 0)
            l = canReach(arr, a);
        if (is_valid(b) && vis[b] == 0)
            r = canReach(arr, b);
        return l || r;
    }
};
