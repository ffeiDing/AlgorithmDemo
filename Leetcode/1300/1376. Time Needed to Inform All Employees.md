class Solution {
public:
    int numOfMinutes(int n, int headID, vector<int>& manager, vector<int>& informTime) {
        unordered_map<int, vector<int>> mp;
        for (int i = 0; i < n; i++){
            if (i != headID){
                mp[manager[i]].push_back(i);
            }
        }
        queue<int> qu;
        queue<int> t;
        int vis[100010];
        memset(vis, 0, sizeof(vis));
        qu.push(headID);
        t.push(0);
        vis[headID] = 1;
        int res = 0;
        while(!qu.empty()){
            int len = qu.size();
            for (int i = 0; i < len; i++){
                int tmp = qu.front();
                int s = t.front();
                qu.pop();
                t.pop();
                if (mp[tmp].size() == 0){
                    res = max(res, s);
                }
                for (auto j : mp[tmp]){
                    if (vis[j] == 0){
                        vis[j] = 1;
                        qu.push(j);
                        t.push(s+informTime[tmp]);
                    }
                }
            }
        }
        return res;
    }
};
