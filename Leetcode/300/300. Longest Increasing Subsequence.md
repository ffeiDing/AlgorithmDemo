class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        int len = nums.size();
        int res[10010];
        res[0] = 1;
        for (int i = 1; i < len; i++){
            res[i] = 1;
            for (int j = 0; j < i; j++){
                if (nums[i] > nums[j]){
                    res[i] = max(res[i], res[j]+1);
                }
            }
        }
        int r = 0;
        for (int i = 0; i < len; i++){
            r = max(r, res[i]);
        }
        return r;
    }
};
