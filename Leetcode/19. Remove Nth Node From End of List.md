/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        int len = 0;
        ListNode* t = head;
        while(t != NULL){
            len++;
            t = t->next;
        }
        int num = len - n;
        if (num == 0)
            return head->next;
        t = head;
        int i = 0;
        while(i!=num-1){
            t = t->next;
            i++;
        }
        ListNode* pre = t;
        t = t->next->next;
        pre->next = t;
        return head;
    }
};
