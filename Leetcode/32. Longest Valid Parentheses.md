class Solution {
public:
    int longestValidParentheses(string s) {
        int len = s.length();
        int dp[100010];
        memset(dp, 0, sizeof(dp));
        for (int i = 0; i < len; i++){
            if (s[i] == ')'){
                if (i-1 >= 0){
                    if (s[i-1] == '(')
                        dp[i]=i-2>=0? (dp[i-2]+2):2;        
                    else{
                        int tmp = i-dp[i-1]-1;
                        if (tmp >= 0 && s[tmp] == '(')
                            dp[i] = tmp-1 >= 0 ? dp[i-1]+dp[tmp-1]+2:dp[i-1]+2;
                    }
                }
            }
        }
        int maxnum = 0;
        for (int i = 0; i < len; i++)
            maxnum = max(maxnum, dp[i]);
        return maxnum;
    }
};

class Solution {
public:
    int longestValidParentheses(string s) {
        int len = s.length();
        stack<int> st;
        int max_len = 0;
        st.push(-1);
        for (int i = 0; i < len; i++){
            if (s[i] == '(')
                st.push(i);
            else{
                if (st.top()!= -1 && s[st.top()]=='('){
                    st.pop();
                    max_len = max(max_len, i-st.top());
                }
                else
                    st.push(i);
            }
        }
        return max_len;
    }
};
