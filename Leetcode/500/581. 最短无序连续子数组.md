class Solution {
public:
    int findUnsortedSubarray(vector<int>& nums) {
        int len = nums.size();
        vector<int> res;
        for (int i = 0; i < len; i++)
            res.push_back(nums[i]);
        sort(res.begin(), res.end());
        int l, r;
        for (l = 0; l < len; l++){
            if (nums[l] != res[l])
                break;
        }
        for (r = len-1; r >= 0; r--){
            if (nums[r] != res[r])
                break;
        }
        if (l == len && r == -1) return 0;
        return (r-l+1);
    }
};
