```
class Solution {
public:
    struct Node{
        int a;
        int b;
        int c;
        int d;
    };
    struct cmp{
        bool operator()(const Node node1, const Node node2) const{
            if (node1.a != node2.a)
                return node1.a < node2.a;
            if (node1.b != node2.b)
                return node1.b < node2.b;
            if (node1.c != node2.c)
                return node1.c < node2.c;
            return node1.d < node2.d;   
        }
    };
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        set<Node, cmp> s;
        vector<vector<int>> res;
        sort(nums.begin(), nums.end());
        int len = nums.size();
        int d = len-1;
        for (int a = 0; a < len; a++){
            for (int b = a+1; b < len; b++){
                int c = b+1;
                if (c > len) continue;
                int d = len-1;
                while(c < d){
                    int temp = nums[a] + nums[b] + nums[c] + nums[d];
                    if (temp == target){
                        Node node;
                        node.a = nums[a];
                        node.b = nums[b];
                        node.c = nums[c];
                        node.d = nums[d];
                        s.insert(node);
                        c++;
                        d--;
                    }
                    else if (temp < target)
                        c++;
                    else
                        d--;
                }
            }
        }
        set<Node>::iterator it;
        for (it = s.begin(); it != s.end(); it++){
            vector<int> tmp;
            tmp.push_back((*it).a);
            tmp.push_back((*it).b);
            tmp.push_back((*it).c);
            tmp.push_back((*it).d);
            res.push_back(tmp);
        }
        return res;
    }
};
```
