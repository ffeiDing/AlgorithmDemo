class Solution {
public:
    int maxArea(vector<int>& height) {
        int len = height.size();
        int res = -1;
        for (int i = 1; i < len; i++){
            int max_num = -1;
            int idx = -1;
            for (int j = 0; j < i; j++){
                int tmp = (i-j)*min(height[j], height[i]);
                if (tmp > max_num){
                    max_num = tmp;
                    idx = j;
                    if (height[j] >= height[i])
                        break;
                }
            }
            res = max(res, max_num);
        }
        return res;
    }
};
