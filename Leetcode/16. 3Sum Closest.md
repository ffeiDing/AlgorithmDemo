class Solution {
public:
    int threeSumClosest(vector<int>& nums, int target) {
        int len = nums.size();
        sort(nums.begin(), nums.end());
        int t = 0;
        while(1){
            for (int i = 0;  i < len; i++){
                int j = i+1;
                int k = len-1;
                int temp = target + t - nums[i];
                while(j < k){
                    if (nums[j] + nums[k] == temp)
                        return target+t;
                    else if (nums[j] + nums[k] < temp)
                        j++;
                    else
                        k--;
                }
            }
            for (int i = 0;  i < len; i++){
                int j = i+1;
                int k = len-1;
                int temp = target - t - nums[i];
                while(j < k){
                    if (nums[j] + nums[k] == temp)
                        return target-t;
                    else if (nums[j] + nums[k] < temp)
                        j++;
                    else
                        k--;
                }
            }
            t++;
        }
    }
};
