class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int len = nums.size();
        int a[100010], b[100010];
        int tmp = 1;
        for (int i = 0; i < len; i++){
            a[i] = tmp*nums[i];
            tmp = a[i];
        }
        tmp = 1;
        for (int i = len-1; i >= 0; i--){
            b[i] = tmp*nums[i];
            tmp = b[i];
        }
        vector<int> res;
        for (int i = 0; i < len; i++){
            if (i==0){
                res.push_back(b[1]);
                continue;
            }
            if (i==len-1){
                res.push_back(a[len-2]);
                continue;
            }
            res.push_back(a[i-1]*b[i+1]);
        }
        return res;
    }
};
