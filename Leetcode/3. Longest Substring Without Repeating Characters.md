class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int len = s.length();
        if (len == 0)
            return 0;
        int num[100010];
        memset(num, 0, sizeof(num));
        num[0] = 1;
        for (int i = 1; i < len; i++){
            num[i] = 1;
            for (int j = i-1; j >= i-num[i-1]; j--){
                if (s[i] != s[j])
                    num[i]++;
                else
                    break;
            }
        }
        int max_res = -1;
        for (int i = 0; i < len; i++)
            max_res = max(max_res, num[i]);
        return max_res;
    }
};
