class Solution {
public:
    class Node {
    public:
        int val;
        Node* nxt;
        Node(int _val){
            val = _val;
            nxt = NULL;
        }
    };
    int find_and_move(Node* head, int num){
        int cnt = 0;
        Node* cur = head;
        Node* pre = head;
        Node* nxt = head;
        while(1){
            pre = cur;
            cur = cur->nxt;
            nxt = cur->nxt;
            if (cur->val == num) break;
            cnt++;
        }
        pre->nxt = nxt;
        cur->nxt = head->nxt;
        head->nxt = cur;
        return cnt;
    }
    vector<int> processQueries(vector<int>& queries, int m) {
        Node* head = new Node(0);
        Node* cur = head;
        for (int i = 1; i <= m; i++){
            cur->nxt = new Node(i);
            cur = cur->nxt;
        }
        for (int i = 0; i < queries.size(); i++){
            int num = queries[i];
            queries[i] = find_and_move(head, num);
        }
        return queries;
    }
};
