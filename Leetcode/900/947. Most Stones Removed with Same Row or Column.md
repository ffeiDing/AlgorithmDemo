class Solution {
public:
    //求连通分量
    int n;
    int vis[1010];
    void dfs(int idx, vector<vector<int>>& stones){
        for (int i = 0; i < n; i++){
            if (vis[i] == 0 && ((stones[i][0] == stones[idx][0]) 
                             || (stones[i][1] == stones[idx][1]))){
                vis[i] = 1;
                dfs(i, stones);
            }
        }
    }
    int removeStones(vector<vector<int>>& stones) {
        n = stones.size();
        int res = n;
        memset(vis, 0, sizeof(vis));
        for (int i = 0; i < n; i++){
            if (vis[i] == 0){
                vis[i] = 1;
                dfs(i, stones);
                res--;
            }
        }
        return res;
    }
};
