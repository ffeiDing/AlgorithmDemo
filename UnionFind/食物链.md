```c
#include <iostream>
#include <iomanip>
#include <cmath>
#include <queue>
#include <cstring>
using namespace std;

int node[50010], res[50010];
int n, k;
int d, x, y;
int sum = 0;

int root(int x, int pre){
	if (x == 0)
		return pre;
	return root(node[x], x);
}

int state(int x){
	int temp = 0;
	while(node[x] != 0){
		temp = (temp + res[x])%3;
		x = node[x];
	}
	//cout << temp << endl;
	return temp;
}

int main(){
	cin >> n >> k;
	memset(node, 0, sizeof(node));
	memset(res, 0, sizeof(res));
	while(k--){
		cin >> d >> x >> y;
		if (x > n || y > n){
			sum++;
			continue;
		}
		int root_x = root(x, -1);
		int root_y = root(y, -1);
		int state_x = state(x);
		int state_y = state(y);
		if (d == 1){
			if (root_x != root_y){
				node[root_y] = root_x;
				res[root_y] = (state_x - state_y + 3)%3;
			}
			else{
				if (state_x != state_y)
					sum++;
			}
		}
		if (d == 2){
			if (x == y){
				sum++;
				continue;
			}
			if (root_x != root_y){
				node[root_y] = root_x;
				res[root_y] = (state_x - state_y + 4)%3;
			}
			else{
				if ((state_x - state_y + 3)%3 != 2)
					sum++;
			}
		}
	}
	cout << sum << endl;
    return 0;
}
```
